45 static int getIPv4 ( const char * dev, char * ipv4 )
46 {
47  struct ifreq ifc;
48  int res;
49  int sockfd = socket ( AF_INET, SOCK_DGRAM, 0 );
50 
51  if ( sockfd < 0 )
52  return -1;
53  strcpy ( ifc.ifr_name, dev );
54  res = ioctl ( sockfd, SIOCGIFADDR, &ifc );

63 Boards_ctrl::Boards_ctrl ( std::string iface, int expected_num_boards = 1 )
64 {
65  int broadcastOn = 1;
66 
67  this->expected_num_boards = expected_num_boards;
68  boards_counter = 0;
69 
70 
71 #ifndef __XENO__
72  char ip[16];
73  if ( getIPv4 ( iface.c_str(), ip ) ) {
74  perror ( "cannot find iface_name" );
75  assert ( 0 );
76  }
77 #endif
78 
79  // create udp socket
80  if ( ( udp_sock = socket ( AF_INET, SOCK_DGRAM, IPPROTO_UDP ) ) < 0 ) {
81  perror ( "socket cannot be created" );
82  assert ( 0 );
83  }
84 
85  // set socket broadcast option
86  setsockopt ( udp_sock, SOL_SOCKET, SO_BROADCAST, &broadcastOn, sizeof ( broadcastOn ) );
87  // set socket timeout option
88 #ifdef __XENO__
89  // This socket control option is used to specify the time-out on the socket before it returns.
90  // It is used typically to wait for data on a Read.
91  // The time-out specifies the amount of time the function will wait for data before it returns.
92  int64_t timeout_ns = 250000000;
93  if ( ioctl ( udp_sock, RTNET_RTIOC_TIMEOUT, &timeout_ns ) < 0 )
94  DPRINTF ( "ioctl RTNET_RTIOC_TIMEOUT failed\n" );
95 #else
96  struct timeval timeout;
97  timeout.tv_sec = 0;
98  timeout.tv_usec = 250000;
99  if ( setsockopt ( udp_sock, SOL_SOCKET, SO_RCVTIMEO, ( char * ) &timeout,
100  sizeof ( timeout ) ) < 0 )
101  DPRINTF ( "setsockopt SO_RCVTIMEO failed\n" );
102 
103  if ( setsockopt ( udp_sock, SOL_SOCKET, SO_SNDTIMEO, ( char * ) &timeout,
104  sizeof ( timeout ) ) < 0 )
105  DPRINTF ( "setsockopt SO_SNDTIMEO failed\n" );
106 #endif
107 
108  // bind the socket to local_addr
109  local_addr.sin_family = AF_INET;
110  local_addr.sin_port = htons ( 0 );
111 #ifdef __XENO__
112  local_addr.sin_addr.s_addr = INADDR_ANY;
113 #else
114  local_addr.sin_addr.s_addr = inet_addr ( ip );
115 #endif
116  if ( bind ( udp_sock, ( struct sockaddr * ) &local_addr, sizeof ( local_addr ) ) < 0 ) {
117  perror ( "cannot bind to local ip/port" );
118  close ( udp_sock );
119  assert ( 0 );
120  }
121 
122  //
123  memset ( ( void* ) &dest_addr, 0, sizeof ( dest_addr ) );
124  dest_addr.sin_family = AF_INET;
125  dest_addr.sin_port = htons ( DEFAULT_PORT );
126  dest_addr.sin_addr.s_addr = INADDR_BROADCAST;
127 }

224 void * Boards_ctrl::rx_udp ( void *_ )
225 {
226 
227  Boards_ctrl * kls = ( Boards_ctrl* ) _;
228 
229  int size;
230  uint8_t buff[RX_UDP_BUFSIZE], computedChecksum;
231 
232 #ifdef __XENO__
233  pthread_set_mode_np ( 0, PTHREAD_WARNSW );
234  pthread_set_name_np ( pthread_self(), "rx_udp" );
235 
236 #endif
237 
238  for ( ;; ) {
239 
240  // I do not use UDPCommPacket because it needs to be modified to receive generic bc data
241  // in this thread we JUST receive udp pkt, we just need to verify the checksum ...
242 
243  size = recv ( kls->udp_sock, &buff, sizeof ( buff ), 0 );
244 
245  if ( size < 0 ) {
246  //DPRINTF("udp recvfrom() %s\n", strerror(errno) );
247  continue;
248  }
249 
250  // verify checksum
251  computedChecksum = 0;
252  for ( int i = 0; i < size-1; i++ ) computedChecksum -= buff[i];
253  if ( computedChecksum != buff[size-1] ) {
254  DPRINTF ( "Checksum mismatch 0x%02X 0x%02X\n", computedChecksum , buff[size-1] );
255  DPRINTF ( "recv %d bytes\n", size );
256  for ( int i=0; i<size; i++ ) {
257  DPRINTF ( "0x%02X ", buff[i] );
258  }
259  DPRINTF ( "\n" );
260  continue;
261  }
262 
263  // switch on command byte
264  switch ( buff[2] ) {
265 
266  case 0x82 :
267  // REPLY_ACTIVE_BOARDS ... same packet for ALL DSP boards !!!
268  // create board instance that handle tcp commands
269  // check buffer overflow and board copies ?!?
270  //DPRINTF("ID %d\n", buff[4]);
271  kls->boards_counter++;//We do not use a mutex here for speed reasons, we know a segfault can happen and will offer you a pizza if you are that lucky
272  kls->factory_board ( buff );
273  break;
274  case 0x84 :
275  kls->pinged_boards[buff[3]]=true;
276  break;
277  case 0xBB :
278  // BCAST_DATA_PACKETS MotorController
279  case 0xBC :
280  // BCAST_DATA_PACKETS ForceTorqueSens
281  case 0xBD :
282  // BCAST_DATA_PACKETS MultiAxis ?!?
283 
284  // process bc_data at Boards_ctrl level
285  kls->on_bc_data ( buff );
286  break;
287 
288  default:
289  break;
290  }
291  }
292 
293  return 0;
294 }
